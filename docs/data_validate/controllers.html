 27class GeneralContext:
 28    """
 29    Base context class for the Data Validate application.
 30
 31    This class serves as a dependency injection container, initializing and holding
 32    references to core services used across the application. It ensures that components
 33    like the logger, configuration, and language manager are instantiated once and
 34    available globally through the context object.
 35
 36    Attributes:
 37        data_args (DataArgs): Command-line arguments and runtime data configurations.
 38        kwargs (Dict[str, Any]): Additional keyword arguments passed during initialization.
 39        lm (LanguageManager): Manager for handling internationalization and localized strings.
 40        config (ApplicationConfig): Central configuration object for application settings and constants.
 41        fs_utils (FileSystemUtils): Utilities for file system operations.
 42        logger_manager (LoggerManager): Manager responsible for configuring the logging system.
 43        logger (logging.Logger): The primary logger instance for the application.
 44        validations_not_run (list): A list to track validations that were skipped or not executed.
 45    """
 46
 47    def __init__(
 48        self,
 49        data_args: DataArgs = None,
 50        **kwargs: Dict[str, Any],
 51    ):
 52        """
 53        Initialize the GeneralContext with a toolkit, configuration, file system utilities, and logger.
 54
 55        Args:
 56            data_args (DataArgs): Data arguments containing input and output folder paths and execution flags.
 57            **kwargs: Additional keyword arguments for extended context configuration.
 58
 59        The initialization process involves:
 60        1.  Storing arguments and kwargs.
 61        2.  Initializing the `LanguageManager` for i18n support.
 62        3.  Loading the application `ApplicationConfig`.
 63        4.  Setting up `FileSystemUtils` for file handling.
 64        5.  Configuring the logging system via `LoggerManager`.
 65        6.  Disabling the logger if debug mode is not active.
 66        """
 67        # Unpack the arguments
 68        self.data_args = data_args
 69        self.kwargs = kwargs
 70
 71        # Configure the Toolkit
 72        self.lm: LanguageManager = LanguageManager()
 73        self.config: ApplicationConfig = ApplicationConfig()
 74        self.fs_utils: FileSystemUtils = FileSystemUtils()
 75        self.logger_manager = LoggerManager(
 76            log_folder="data/output/logs",
 77            console_logger="console_logger",
 78            prefix="data_validate",
 79            logger_name="data_validate_file_logger",
 80        )
 81        self.logger = self.logger_manager.file_logger
 82
 83        # Configure the file logger
 84        if not self.data_args.data_action.debug:
 85            self.logger.disabled = True
 86
 87        self.validations_not_run = []

 90        """
 91        Finalize the application context and cleanup resources.
 92
 93        This method handles post-execution tasks such as removing temporary log files
 94        if the application is not running in debug mode.
 95
 96        Logic:

 98            - If debug mode is ON: Prints the location of the log file to stdout.
 99        """

 82    """
 83    Data model for a collection of validation reports, accessible by test name.
 84
 85    Aggregates multiple `ModelItemReport` instances and provides utility methods
 86    to report and manipulate validation results globally.
 87
 88    Attributes:
 89        context (Optional[GeneralContext]): Application context for localization and utils.
 90        reports (dict[str, ModelItemReport]): Dictionary of reports indexed by their test name.
 91    """
 92
 93    def __init__(self, context: Optional["GeneralContext"] = None, reports: Optional[List[ModelItemReport]] = None):
 94        """
 95        Initialize a ModelListReport.
 96
 97        Args:
 98            context (Optional[GeneralContext]): DI context. Defaults to None.
 99            reports (Optional[List[ModelItemReport]]): Initial list of reports. Defaults to None.
100        """
101        self.context = context
102        self.reports: dict[str, ModelItemReport] = {}
103        if reports:
104            for report in reports:
105                self.add_report(report)
106
107    def set_not_executed(self, name_test: str) -> None:
108        """
109        Mark a specific test as not executed.
110
111        If the report does not exist, it creates a new one marked as not executed.
112
113        Args:
114            name_test (str): The name of the test to mark.
115        """
116        if name_test in self.reports:
117            self.reports[name_test].was_executed = False
118        else:
119            self.reports[name_test] = ModelItemReport(name_test)
120            self.reports[name_test].was_executed = False

123        """

186    def flatten(self, n_messages: int, locale: str = "pt_BR") -> "ModelListReport":
187        """
188        Create a new ModelListReport with truncated error/warning lists.
189
190        Limits the number of messages per report to `n_messages`. If messages are
191        truncated, adds a summary message indicating count of omitted items.
192
193        Args:
194            n_messages (int): Maximum number of messages to retain per category.
195            locale (str): Locale for message formatting. Defaults to "pt_BR".
196
197        Returns:
198            ModelListReport: A new flattened report instance.
199        """
200        flattened_reports = []
201        for report in self.reports.values():
202            flattened_report = ModelItemReport(
203                name_test=report.name_test,
204                errors=report.errors[:n_messages],
205                warnings=report.warnings[:n_messages],
206            )
207            if len(report.errors) > n_messages:
208                count_omitted_errors = NumberFormattingProcessing.format_number_brazilian(len(report.errors) - n_messages, locale)
209                flattened_report.add_error(self.context.lm.text("model_report_msg_errors_omitted", count=count_omitted_errors))
210
211            if len(report.warnings) > n_messages:
212                count_omitted_warnings = NumberFormattingProcessing.format_number_brazilian(len(report.warnings) - n_messages, locale)
213                flattened_report.add_warning(
214                    self.context.lm.text(
215                        "model_report_msg_warnings_omitted",
216                        count=count_omitted_warnings,
217                    )
218                )
219
220            flattened_reports.append(flattened_report)
221        return ModelListReport(context=self.context, reports=flattened_reports)

161            # Dynamically create the attribute name, e.g., "sp_description"

163
164            # Model instance creation and initialization
165            model_instance = model_class(
166                context=self.context,
167                data_model=self.all_load_data.get(sp_name_key),
168                **self.kwargs,
169            )
170            setattr(self, attribute_name, model_instance)
171            self.models_to_use.append(model_instance)
172
173            self.report_list.extend(
174                self.TITLES_INFO[config.NamesEnum.FS.value],
175                errors=model_instance.structural_errors,
176                warnings=model_instance.structural_warnings,
177            )
178            self.report_list.extend(
179                self.TITLES_INFO[config.NamesEnum.FC.value],
180                errors=model_instance.data_cleaning_errors,
181                warnings=model_instance.data_cleaning_warnings,
182            )
183
184            if FLAG is not None:
185                self.context.logger.info(f"Initialized model: {attribute_name} = {model_instance}")

196        4. **Optional**: Secondary business rules (Values, Scenarios, Legends).
197        """
198        self.context.logger.info("Building validation pipeline...")
199

203        # RUN ALL VALIDATIONS PIPELINE

208        # 2. Validate the spelling of the data

215        validators.SpTemporalReferenceValidator(data_models_context=self.data_models_context, report_list=self.report_list)

217        # 4. Validate spreadsheet data optional

219        validators.SpValueValidator(data_models_context=self.data_models_context, report_list=self.report_list)

224        """
225        Generate final validation reports.
226
227        Compiles all collected errors and warnings into `ModelListReport`,
228        logs debug information (if enabled), calculates summary statistics,
229        and triggers the generation of HTML/PDF reports via `ReportGeneratorFiles`.
230        """
231        self.context.logger.info("Generating reports...")

234            self.context.logger.info("\nModo DEBUG ativado.")
235            self.context.logger.info("------ Resultados da verificao dos testes ------")
236
237            for report in self.report_list:
238                self.context.logger.info(f"Report: {report.name_test}")
239                self.context.logger.error(f"  Errors: {len(report.errors)}")
240                for error in report.errors:
241                    self.context.logger.error(f"    - {error}")
242                self.context.logger.warning(f"  Warnings: {len(report.warnings)}")
243                for warning in report.warnings:
244                    self.context.logger.warning(f"    - {warning}")
245                self.context.logger.info("---------------------------------------------------------------")
246
247        # Set summary of total errors and warnings
248        total_errors = sum(len(report.errors) for report in self.report_list)
249        total_warnings = sum(len(report.warnings) for report in self.report_list)
250
251        if self.context.data_args.data_action.debug:
252            self.context.logger.error(f"Total errors: {total_errors}")
253            self.context.logger.warning(f"Total warnings: {total_warnings}")
254
255        # Generate report in HTML and PDF formats
256        controllers.ReportGeneratorFiles(context=self.context).build_report(report_list=self.report_list)
