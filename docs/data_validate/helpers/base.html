 25    def detect_encoding(self, file_path: str, num_bytes: int = 1024) -> Tuple[bool, str]:
 26        """
 27        Detects the encoding of a file by reading a specified number of bytes.
 28
 29        Args:
 30            file_path (str): The path to the file.
 31            num_bytes (int): The number of bytes to read for encoding detection. Default is 1024.
 32
 33        Returns:
 34            Tuple[bool, str]: A tuple containing a boolean indicating success or failure
 35                              and the detected encoding or a translated error message.
 36        """
 37        try:
 38            if not file_path:
 39                return False, self.lm.text("fs_utils_error_file_path_empty")
 40            if not os.path.exists(file_path):
 41                return False, self.lm.text(
 42                    "fs_utils_error_file_not_found",
 43                    filename=os.path.basename(file_path),
 44                )
 45            if not os.path.isfile(file_path):
 46                return False, self.lm.text("fs_utils_error_path_not_file", path=file_path)
 47
 48            with open(file_path, "rb") as f:
 49                raw_data = f.read(num_bytes)
 50                result = chardet.detect(raw_data)
 51                encoding = result.get("encoding")
 52                if not encoding:
 53                    return False, self.lm.text("fs_utils_error_encoding_failed")
 54                return True, encoding
 55        except OSError as e:
 56            return False, self.lm.text("fs_utils_error_encoding_os", error=str(e))
 57        except Exception as e:
 58            return False, self.lm.text("fs_utils_error_unexpected", error=str(e))

 65    def configure_logger(
 66        self,
 67        logger_name: str,
 68        level: Optional[int] = None,
 69        log_file: Optional[str] = None,
 70    ) -> logging.Logger:
 71        """
 72        Configures a logger with the specified name, level, and optional log file.
 73
 74        Args:
 75            logger_name (str): The name of the logger.
 76            level (Optional[int]): The logging level. If not provided, the default level is used.
 77            log_file (Optional[str]): The path to the log file. If provided, logs will also be written to this file.
 78
 79        Returns:
 80            logging.Logger: The configured logger instance.
 81        """
 82        logger = logging.getLogger(logger_name)
 83        logger.setLevel(level or self.default_level)
 84        logger.propagate = False
 85
 86        if logger.hasHandlers():
 87            logger.handlers.clear()
 88
 89        # Console handler
 90        console_handler = logging.StreamHandler()
 91
 92        # Formatter for log messages
 93        console_handler.setFormatter(CustomFormatter())
 94        logger.addHandler(console_handler)
 95
 96        # File handler (if log_file is provided)
 97        if log_file:
 98            file_handler = logging.FileHandler(log_file)
 99            file_handler.setFormatter(CustomFormatter())
100            logger.addHandler(file_handler)
101
102        return logger
