 40    def detect_encoding(self, file_path: str, num_bytes: int = 1024) -> Tuple[bool, str]:
 41        """
 42        Detect file encoding by reading a sample of bytes.
 43
 44        Reads the first `num_bytes` of a file and uses `chardet` to guess its
 45        character encoding.
 46
 47        Args:
 48            file_path (str): Absolute or relative path to the file.
 49            num_bytes (int, optional): Number of bytes to read for detection. Defaults to 1024.
 50
 51        Returns:
 52            Tuple[bool, str]:
 53                - bool: True if detection was successful, False otherwise.
 54                - str: The detected encoding name (e.g., 'utf-8') or a localized error message.
 55        """
 56        try:
 57            if not file_path:
 58                return False, self.lm.text("fs_utils_error_file_path_empty")
 59            if not os.path.exists(file_path):
 60                return False, self.lm.text(
 61                    "fs_utils_error_file_not_found",
 62                    filename=os.path.basename(file_path),
 63                )
 64            if not os.path.isfile(file_path):
 65                return False, self.lm.text("fs_utils_error_path_not_file", path=file_path)
 66
 67            with open(file_path, "rb") as f:
 68                raw_data = f.read(num_bytes)
 69                result = chardet.detect(raw_data)
 70                encoding = result.get("encoding")
 71                if not encoding:
 72                    return False, self.lm.text("fs_utils_error_encoding_failed")
 73                return True, encoding
 74        except OSError as e:
 75            return False, self.lm.text("fs_utils_error_encoding_os", error=str(e))
 76        except Exception as e:
 77            return False, self.lm.text("fs_utils_error_unexpected", error=str(e))

 80    def __init__(
 81        self,
 82        log_folder: str = "logs",
 83        default_level: int = logging.DEBUG,
 84        console_logger="console_logger",
 85        prefix="data_validate",
 86        logger_name="data_validate_file_logger",
 87    ):
 88        """
 89        Initialize the LoggerManager.
 90
 91        Sets up the log directory and initializes both the console and file loggers.
 92
 93        Args:
 94            log_folder (str, optional): Directory for log files. Defaults to "logs".
 95            default_level (int, optional): Default log level. Defaults to logging.DEBUG.
 96            console_logger (str, optional): Name for the console logger. Defaults to "console_logger".
 97            prefix (str, optional): Prefix for the log file name. Defaults to "data_validate".
 98            logger_name (str, optional): Name for the file logger. Defaults to "data_validate_file_logger".
 99        """
100        self.log_folder = log_folder
101        self.default_level = default_level
102        if not os.path.exists(self.log_folder):
103            os.makedirs(self.log_folder)
104
105        self.console_logger = self.configure_logger(console_logger)
106        self.log_file = self.generate_log_file_name(prefix=prefix)
107        self.file_logger = self.configure_logger(logger_name=logger_name, log_file=self.log_file)

109    def configure_logger(
110        self,
111        logger_name: str,
112        level: Optional[int] = None,
113        log_file: Optional[str] = None,
114    ) -> logging.Logger:
115        """
116        Configure a logger with the specified name, level, and optional output file.
117
118        Creates or retrieves a logger, clears existing handlers to prevent duplication,
119        and attaches a console handler (with colors) and optionally a file handler.
120
121        Args:
122            logger_name (str): The unique name of the logger.
123            level (Optional[int]): The logging level. Uses default_level if None.
124            log_file (Optional[str]): Path to a file where logs should be saved.
125
126        Returns:
127            logging.Logger: The fully configured logger instance.
128        """
129        logger = logging.getLogger(logger_name)
130        logger.setLevel(level or self.default_level)
131        logger.propagate = False
132
133        if logger.hasHandlers():
134            logger.handlers.clear()
135
136        # Console handler
137        console_handler = logging.StreamHandler()
138
139        # Formatter for log messages
140        console_handler.setFormatter(CustomFormatter())
141        logger.addHandler(console_handler)
142
143        # File handler (if log_file is provided)
144        if log_file:
145            file_handler = logging.FileHandler(log_file)
146            file_handler.setFormatter(CustomFormatter())
147            logger.addHandler(file_handler)
148
149        return logger
