 93    def load_all(self):
 94        errors = []
 95        files_map, qml_files, missing_files = self.scanner.scan()
 96
 97        data = {}
 98        for name, path in files_map.items():
 99            _, header_type, _ = self.config.file_specs[name]
100            if header_type == "single":
101                strat = SingleHeaderStrategy()
102            elif header_type == "double":
103                strat = DoubleHeaderStrategy()
104            else:
105                # qml will not pass through here
106                continue
107            reader = ReaderFactory.get_reader(path, strat)
108
109            # Configure DataModel
110            df_local = None
111            try:
112                df_local = reader.read()
113            except FileNotFoundError as e:
114                errors.append(f"{path.name}: Arquivo não encontrado no diretório. Detalhes: {e} ({type(e)})")
115            except UnicodeDecodeError as e:
116                errors.append(f"{path.name}: Erro de codificação do arquivo. Verifique se está em UTF-8. Detalhes: {e} ({type(e)})")
117            except pd.errors.ParserError as e:
118                errors.append(
119                    f"{path.name}: Erro na estrutura da planilha. Verifique se há células mescladas ou formato inválido. Detalhes: {e} ({type(e)})"
120                )
121            except ValueError as e:
122                errors.append(f"{path.name}: Erro nos valores da planilha. Verifique se os tipos de dados estão corretos. Detalhes: {e} ({type(e)})")
123            except IOError as e:
124                errors.append(
125                    f"{path.name}: Erro de entrada/saída ao ler o arquivo. Verifique se ele não está aberto em outro programa. Detalhes: {e} ({type(e)})"
126                )
127            except Exception as e:
128                errors.append(f"{path.name}: Erro inesperado ao processar o arquivo. Detalhes: {e} ({type(e)})")
129
130            data_model = DataLoaderModel(
131                input_folder=str(self.input_dir),
132                path=path,
133                raw_data=df_local if df_local is not None else pd.DataFrame(),
134                is_read_successful=True if df_local is not None else False,
135            )
136
137            data[name] = data_model
138
139        # Add raw QMLs
140        data["qmls"] = [ReaderFactory.get_reader(q, SingleHeaderStrategy()).read() for q in qml_files]
141
142        # Add missing or non-required files as empty
143        for name, (req, _, _) in self.config.file_specs.items():
144            if name not in data:
145                data[name] = DataLoaderModel(
146                    input_folder=str(self.input_dir),
147                    path=Path(name),
148                    raw_data=pd.DataFrame(),
149                    is_read_successful=False,
150                )
151
152        return data, errors

 11class LanguageManager:
 12    """
 13    A class to manage localization and translations for the application.
 14    """
 15
 16    def __init__(self, path_locale_dir=None):
 17        """
 18        Initializes the LanguageManager with the given locale directory and default language.
 19
 20        Args:
 21            path_locale_dir (str): Path to the directory containing locale files.
 22        """
 23        self.path_locale_dir = path_locale_dir or Path(__file__).resolve().parents[3] / "static" / "locales"
 24
 25        self.default_language = LanguageEnum.DEFAULT_LANGUAGE.value
 26        self.current_language = None
 27        self.translations = {}
 28
 29        # Pipiline to configure the language
 30        self._congifure_language()
 31        self.supported_languages = LanguageEnum.list_supported_languages()
 32        self._load_translations(self.current_language)
 33
 34    def _congifure_language(self):
 35        store_locale_path = Path(__file__).resolve().parents[4] / ".config" / "store.locale"
 36
 37        # Check if the store.locale file exists
 38        if not store_locale_path.exists():
 39            print(f"store.locale not found. Falling back to default language '{self.default_language}'.")
 40            self.current_language = self.default_language
 41
 42            # Create the store.locale file with the default language
 43            store_locale_path.parent.mkdir(parents=True, exist_ok=True)
 44            with store_locale_path.open("w", encoding="utf-8") as f:
 45                f.write(self.default_language)
 46            return
 47
 48        with store_locale_path.open("r", encoding="utf-8") as f:
 49            self.current_language = f.read().strip()
 50            if self.current_language not in LanguageEnum.list_supported_languages():
 51                print(f"Invalid current language '{self.current_language}' in store.locale. Falling back to '{LanguageEnum.DEFAULT_LANGUAGE.value}'.")
 52                self.current_language = self.default_language
 53
 54    def _load_translations(self, lang_code):
 55        """Loads translations from the JSON file for a given language."""
 56        filepath = os.path.join(self.path_locale_dir, lang_code, "messages.json")
 57        try:
 58            with open(filepath, "r", encoding="utf-8") as f:
 59                self.translations = json.load(f)
 60            return True
 61        except FileNotFoundError:
 62            print(f"WARNING: Translation file not found for language '{lang_code}': {filepath}")
 63        except json.JSONDecodeError:
 64            print(f"ERROR: Could not decode JSON for language '{lang_code}': {filepath}")
 65        except Exception as e:
 66            print(f"ERROR: Unexpected error loading language '{lang_code}': {e}")
 67        return False
 68
 69    def set_language(self, lang_code):
 70        """
 71        Sets the current language and loads the corresponding translations.
 72
 73        Args:
 74            lang_code (str): Language code to set.
 75
 76        Returns:
 77            bool: True if the language was successfully set, False otherwise.
 78        """
 79        if lang_code in self.supported_languages:
 80            if self._load_translations(lang_code):
 81                self.current_language = lang_code
 82                return True
 83            else:
 84                print(
 85                    self.text(
 86                        "lang_load_error",
 87                        lang=lang_code,
 88                        default_lang=self.default_language,
 89                    )
 90                )
 91                if self._load_translations(self.default_language):
 92                    self.current_language = self.default_language
 93                    return False
 94                return None
 95        else:
 96            print(self.text("invalid_language", default_lang=self.default_language))
 97            if self.current_language != self.default_language:
 98                if self._load_translations(self.default_language):
 99                    self.current_language = self.default_language
100            return False
101
102    def text(self, key, **kwargs):
103        """
104        Retrieves the translated string for the given key in the current language.
105
106        Args:
107            key (str): The key of the string to translate.
108            **kwargs: Keyword arguments for string formatting.
109
110        Returns:
111            str: The translated and formatted string, or a fallback string.
112        """
113        translation_object = self.translations.get(key)
114        if isinstance(translation_object, dict):
115            text = translation_object.get("message", f"<Message for '{key}' missing in '{self.current_language}'>")
116        else:
117            text = f"<'{key}' missing or invalid structure in '{self.current_language}'>"
118
119        try:
120            return text.format(**kwargs) if kwargs else text
121        except KeyError as e:
122            print(f"Warning: Formatting error for key '{key}'. Missing placeholder: {e}")
123            return text
124        except Exception as format_exc:
125            print(f"Warning: Generic formatting error for key '{key}': {format_exc}")
126            return text
127
128    def get_supported_languages(self):
129        """Returns the list of supported language codes."""
130        return self.supported_languages
131
132    def get_current_language(self):
133        """Returns the currently selected language code."""
134        return self.current_language
135
136    def get_info(self):
137        """
138        Returns the current language and supported languages.
139        """
140        return {
141            "current_language": self.current_language,
142            "supported_languages": self.supported_languages,
143        }
144
145    def __str__(self):
146        """
147        Returns a string representation of the LanguageManager object.
148        """
149        return f"LanguageManager(current_language={self.current_language}, supported_languages={self.supported_languages})"
