110    def get_verify_names(self):
111        """
112        Retrieves a dictionary of verification names with localized text.
113
114        The method iterates over `NamesEnum` members and retrieves their localized
115        string representation using the `LanguageManager`. For specific verifications
116        like `TITLES_N` and `SIMP_DESC_N`, it formats the message with dynamic values
117        from configuration constants.
118
119        Returns:
120            MappingProxyType: A read-only dictionary mapping verification keys (from `NamesEnum`)
121            to their localized text descriptions.
122        """
123        keys = [element for element in NamesEnum]
124        values = [
125            (
126                self.lm.text(
127                    str(element.value),
128                    value=(
129                        self.TITLE_OVER_N_CHARS
130                        if element == NamesEnum.TITLES_N
131                        else (self.SIMPLE_DESCRIPTIONS_OVER_N_CHARS if element == NamesEnum.SIMP_DESC_N else None)
132                    ),
133                )
134                if element in (NamesEnum.TITLES_N, NamesEnum.SIMP_DESC_N)
135                else self.lm.text(str(element.value))
136            )
137            for element in keys
138        ]
139        return MappingProxyType(dict(zip([e.value for e in keys], values)))
